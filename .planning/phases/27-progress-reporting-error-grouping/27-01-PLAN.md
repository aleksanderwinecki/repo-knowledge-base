```typescript
// src/indexer/progress.ts — to be created

export type ErrorCategory = 'worktree_conflict' | 'dirty_tree' | 'timeout' | 'no_branch' | 'other';

export interface CollectedError {
  repo: string;
  category: ErrorCategory;
  message: string;
}

export interface PipelineCallbacks {
  progress?: ProgressReporter;
  errors?: ErrorCollector;
}

export class ProgressReporter {
  constructor(stream?: NodeJS.WriteStream);
  update(current: number, total: number, label?: string): void;
  finish(message?: string): void;
}

export class ErrorCollector {
  addRefreshError(repo: string, errorMsg: string): void;
  addNoBranch(repo: string): void;
  addIndexError(repo: string, errorMsg: string): void;
  hasErrors(): boolean;
  printSummary(stream: NodeJS.WriteStream): void;
}
```

Git error strings to classify (from src/indexer/git.ts):
- `'dirty working tree, skipping checkout'` → dirty_tree
- Error messages containing `'ETIMEDOUT'` or `'SIGTERM'` → timeout
- Error messages containing `'already checked out at'` or `'is locked'` → worktree_conflict
- Everything else → other
