```typescript
import type Database from 'better-sqlite3';
import type { LearnedFact } from './types.js';

export function learnFact(db: Database.Database, content: string, repo?: string): LearnedFact {
  const stmt = db.prepare('INSERT INTO learned_facts (content, repo) VALUES (?, ?)');
  const result = stmt.run(content, repo ?? null);
  const id = Number(result.lastInsertRowid);

  // Index in FTS for searchability (KNOW-02)
  db.prepare(
    'INSERT INTO knowledge_fts (name, description, entity_type, entity_id) VALUES (?, ?, ?, ?)'
  ).run(content, content, 'learned_fact', id);

  return {
    id,
    content,
    repo: repo ?? null,
    createdAt: new Date().toISOString(),
  };
}

export function listFacts(db: Database.Database, repo?: string): LearnedFact[] {
  const sql = repo
    ? 'SELECT id, content, repo, created_at FROM learned_facts WHERE repo = ? ORDER BY created_at DESC'
    : 'SELECT id, content, repo, created_at FROM learned_facts ORDER BY created_at DESC';
  const params = repo ? [repo] : [];
  const rows = db.prepare(sql).all(...params) as Array<{
    id: number; content: string; repo: string | null; created_at: string;
  }>;
  return rows.map(r => ({
    id: r.id,
    content: r.content,
    repo: r.repo,
    createdAt: r.created_at,
  }));
}

export function forgetFact(db: Database.Database, id: number): boolean {
  // Remove from FTS first
  db.prepare('DELETE FROM knowledge_fts WHERE entity_type = ? AND entity_id = ?')
    .run('learned_fact', id);
  // Remove from table
  const result = db.prepare('DELETE FROM learned_facts WHERE id = ?').run(id);
  return result.changes > 0;
}
```

NOTE: We bypass the `indexEntity` helper from fts.ts and write directly to the FTS table. This avoids type-system issues since EntityType doesn't include 'learned_fact', while the FTS table's TEXT column accepts any string.

```typescript
import type { Command } from '@commander-js/extra-typings';
import { withDb } from '../db.js';
import { forgetFact } from '../../knowledge/store.js';
import { output, outputError } from '../output.js';

export function registerForget(program: Command) {
  program
    .command('forget')
    .description('Delete a learned fact by ID')
    .argument('<id>', 'fact ID to delete')
    .action((idStr) => {
      const id = parseInt(idStr, 10);
      if (isNaN(id)) {
        outputError('Invalid fact ID — must be a number', 'INVALID_ID');
      }
      const deleted = withDb((db) => forgetFact(db, id));
      if (!deleted) {
        outputError(`Fact ${id} not found`, 'NOT_FOUND');
      }
      output({ deleted: true, id });
    });
}
```

```typescript
function hydrateLearnedFact(db: Database.Database, match: FtsMatch): TextSearchResult | null {
  const row = db
    .prepare('SELECT id, content, repo, created_at FROM learned_facts WHERE id = ?')
    .get(match.entity_id) as { id: number; content: string; repo: string | null; created_at: string } | undefined;

  if (!row) return null;

  return {
    entityType: 'learned_fact' as EntityType,  // Cast needed — FTS returns this string
    entityId: match.entity_id,
    name: row.content.substring(0, 100),  // First 100 chars as name
    snippet: row.content,
    repoName: row.repo ?? 'user-knowledge',  // Default repo name for unassociated facts
    repoPath: '',
    filePath: null,
    relevance: match.relevance,
  };
}
```

This ensures `kb search 'billing domain'` returns learned facts alongside indexed content.
