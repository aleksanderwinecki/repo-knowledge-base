```typescript
export function output(data: unknown): void {
  process.stdout.write(JSON.stringify(data, null, 2) + '\n');
}

export function outputError(message: string, code?: string): never {
  process.stderr.write(JSON.stringify({ error: message, code: code ?? 'ERROR' }, null, 2) + '\n');
  process.exit(1);
}
```

NOTE: Per user decision, JSON-only output, no colors, no human formatting. Use `process.stdout.write` not `console.log` to avoid extra newline behavior differences. Error output goes to stderr so it doesn't corrupt JSON on stdout.

3. Create `src/cli/db.ts` — database path resolution and lifecycle helper:
```typescript
import os from 'os';
import path from 'path';
import { openDatabase, closeDatabase } from '../db/database.js';
import type Database from 'better-sqlite3';

export function getDbPath(): string {
  return process.env.KB_DB_PATH ?? path.join(os.homedir(), '.kb', 'knowledge.db');
}

export function withDb<T>(fn: (db: Database.Database) => T): T {
  const db = openDatabase(getDbPath());
  try {
    return fn(db);
  } finally {
    closeDatabase(db);
  }
}
```

The `withDb` helper ensures DB is always closed, even on exceptions. Every command uses this.

1. Create `src/cli/commands/index-cmd.ts`:

```typescript
import type { Command } from '@commander-js/extra-typings';
import { withDb } from '../db.js';
import { indexAllRepos } from '../../indexer/pipeline.js';
import { output } from '../output.js';
import os from 'os';
import path from 'path';

export function registerIndex(program: Command) {
  program
    .command('index')
    .description('Index all repos under root directory')
    .option('--root <path>', 'root directory to scan', path.join(os.homedir(), 'Documents', 'Repos'))
    .option('--force', 'force re-index all repos', false)
    .action((opts) => {
      const results = withDb((db) =>
        indexAllRepos(db, { rootDir: opts.root, force: opts.force })
      );
      output(results);
    });
}
```

NOTE on naming: File is `index-cmd.ts` (not `index.ts`) to avoid collision with barrel exports. The command name is `index`.

2. Create `src/cli/commands/search.ts`:
```typescript
import type { Command } from '@commander-js/extra-typings';
import { withDb } from '../db.js';
import { searchText } from '../../search/index.js';
import { findEntity } from '../../search/index.js';
import { output, outputError } from '../output.js';
import type { EntityType } from '../../types/entities.js';

export function registerSearch(program: Command) {
  program
    .command('search')
    .description('Full-text search across indexed knowledge')
    .argument('<query>', 'search query (FTS5 syntax supported)')
    .option('--repo <name>', 'filter by repo name')
    .option('--type <entity>', 'filter by entity type (repo, module, event, service)')
    .option('--limit <n>', 'max results', '20')
    .option('--entity', 'structured entity query mode (returns entity cards with relationships)')
    .action((query, opts) => {
      withDb((db) => {
        if (opts.entity) {
          const cards = findEntity(db, query, {
            type: opts.type as EntityType | undefined,
            repo: opts.repo,
          });
          output(cards);
        } else {
          const results = searchText(db, query, {
            limit: parseInt(opts.limit),
            repoFilter: opts.repo,
            entityTypeFilter: opts.type as EntityType | undefined,
          });
          output(results);
        }
      });
    });
}
```

The `--entity` flag switches to structured entity query mode. This keeps a single `search` command with two modes rather than separate commands.

4. Create `src/cli/commands/status.ts`:

Returns database summary stats: number of repos, modules, events, services, edges, learned facts (if table exists).

```typescript
import type { Command } from '@commander-js/extra-typings';
import { withDb } from '../db.js';
import { output } from '../output.js';
import { getDbPath } from '../db.js';

export function registerStatus(program: Command) {
  program
    .command('status')
    .description('Show knowledge base status and statistics')
    .action(() => {
      const dbPath = getDbPath();
      const stats = withDb((db) => {
        const count = (table: string) => {
          const row = db.prepare(`SELECT COUNT(*) as n FROM ${table}`).get() as { n: number };
          return row.n;
        };

        // Check if learned_facts table exists (may not exist until Plan 02 migration)
        let learnedFacts = 0;
        try {
          learnedFacts = count('learned_facts');
        } catch {
          // Table doesn't exist yet
        }

        return {
          database: dbPath,
          repos: count('repos'),
          files: count('files'),
          modules: count('modules'),
          services: count('services'),
          events: count('events'),
          edges: count('edges'),
          learnedFacts,
        };
      });
      output(stats);
    });
}
```
