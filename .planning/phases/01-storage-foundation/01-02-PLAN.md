```typescript
export function tokenizeForFts(text: string): string {
  if (!text) return '';
  return text
    .replace(/([a-z0-9])([A-Z])/g, '$1 $2')       // camelCase split
    .replace(/([A-Z]+)([A-Z][a-z])/g, '$1 $2')     // acronym split (HTTPSServer -> HTTPS Server)
    .replace(/[_./\\]/g, ' ')                        // separators to spaces
    .replace(/[^a-zA-Z0-9\s]/g, ' ')                // other non-alphanumeric to spaces
    .replace(/\s+/g, ' ')                            // collapse spaces
    .trim()
    .toLowerCase();
}
```
